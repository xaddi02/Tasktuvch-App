import React, { useState, useEffect } from 'react';
import { Users, LogOut, Plus, Edit2, Trash2, Shield, AlertCircle, Search, Plane, Lock, FileText, Database, Activity, ShoppingCart, CheckCircle } from 'lucide-react';

// --- SIMULACIÓN DE SEGURIDAD (CRIPTOGRAFÍA BÁSICA PARA DEMOSTRACIÓN) ---
const hashData = (str) => btoa(str).split('').reverse().join('').substring(0, 15);
const encryptData = (str) => `ENC_${btoa(str)}`;
const decryptData = (str) => str.startsWith('ENC_') ? atob(str.replace('ENC_', '')) : str;

const ROLES = {
  ADMIN: 'Admin',
  VENDEDOR: 'Vendedor',
  AUDITOR: 'Auditor'
};

const MOCK_USERS = {
  'admin': { username: 'admin', passwordHash: hashData('123'), role: ROLES.ADMIN },
  'vendedor': { username: 'vendedor', passwordHash: hashData('123'), role: ROLES.VENDEDOR },
  'auditor': { username: 'auditor', passwordHash: hashData('123'), role: ROLES.AUDITOR },
};

export default function App() {
  const [currentUser, setCurrentUser] = useState(null);

  if (!currentUser) {
    return <LoginScreen onLogin={setCurrentUser} />;
  }

  return <Dashboard user={currentUser} onLogout={() => setCurrentUser(null)} />;
}

// --- PANTALLA DE LOGIN CON CAPTCHA Y PRIVACIDAD ---
function LoginScreen({ onLogin }) {
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [error, setError] = useState('');
  const [captcha, setCaptcha] = useState({ a: 0, b: 0, input: '' });
  const [privacyAccepted, setPrivacyAccepted] = useState(false);
  const [showPrivacyModal, setShowPrivacyModal] = useState(false);

  const generateCaptcha = () => {
    setCaptcha({ a: Math.floor(Math.random() * 10) + 1, b: Math.floor(Math.random() * 10) + 1, input: '' });
  };

  useEffect(() => { generateCaptcha(); }, []);

  const handleLogin = (e) => {
    e.preventDefault();
    setError('');

    if (!privacyAccepted) {
      setError('Debes aceptar el Aviso de Privacidad para continuar.');
      return;
    }

    if (parseInt(captcha.input) !== (captcha.a + captcha.b)) {
      setError('El CAPTCHA es incorrecto. Por favor, intenta de nuevo.');
      generateCaptcha();
      return;
    }

    if (!username || !password) {
      setError('Por favor, ingresa usuario y contraseña.');
      return;
    }
    
    const user = MOCK_USERS[username.toLowerCase()];
    if (user && user.passwordHash === hashData(password)) {
      onLogin(user);
    } else {
      setError('Credenciales incorrectas o usuario no encontrado.');
      generateCaptcha();
    }
  };

  return (
    <div className="min-h-screen bg-slate-100 flex items-center justify-center p-4">
      <div className="bg-white rounded-xl shadow-xl max-w-md w-full p-8 relative overflow-hidden">
        <div className="flex flex-col items-center mb-6">
          <div className="bg-blue-600 p-3 rounded-full text-white mb-4 shadow-md">
            <Lock size={32} />
          </div>
          <h1 className="text-2xl font-bold text-slate-800">AeroCRM Seguro</h1>
          <p className="text-slate-500 text-sm mt-1 text-center">Acceso corporativo</p>
        </div>

        {error && (
          <div className="bg-red-50 text-red-600 p-3 rounded-lg flex items-start gap-2 mb-6 text-sm border border-red-100">
            <AlertCircle size={18} className="shrink-0 mt-0.5" />
            <p>{error}</p>
          </div>
        )}

        <form onSubmit={handleLogin} className="space-y-4">
          <div>
            <label className="block text-sm font-medium text-slate-700 mb-1">Usuario</label>
            <input 
              type="text" 
              value={username} 
              onChange={(e) => setUsername(e.target.value)}
              list="usuarios-sugeridos"
              className="w-full px-4 py-2 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none" 
              placeholder="Escribe para buscar (ej. ad...)" 
            />
            <datalist id="usuarios-sugeridos">
              <option value="admin" />
              <option value="vendedor" />
              <option value="auditor" />
            </datalist>
          </div>
          <div>
            <label className="block text-sm font-medium text-slate-700 mb-1">Contraseña</label>
            <input 
              type="password" 
              value={password} 
              onChange={(e) => setPassword(e.target.value)}
              className="w-full px-4 py-2 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none" 
              placeholder="••••••••" 
            />
          </div>

          <div className="bg-slate-50 p-3 rounded-lg border border-slate-200 flex items-center justify-between">
            <span className="text-sm font-medium text-slate-700">CAPTCHA: ¿Cuánto es {captcha.a} + {captcha.b}?</span>
            <input 
              type="number" 
              value={captcha.input} 
              onChange={(e) => setCaptcha({ ...captcha, input: e.target.value })}
              className="w-20 px-2 py-1 border border-slate-300 rounded text-center outline-none focus:ring-2 focus:ring-blue-500" 
            />
          </div>

          <div className="bg-blue-50/50 p-3 rounded-lg border border-blue-100 space-y-2">
             <div className="flex items-start gap-2">
                <input 
                  type="checkbox" 
                  id="privacy" 
                  checked={privacyAccepted} 
                  onChange={(e) => setPrivacyAccepted(e.target.checked)}
                  className="mt-1 shrink-0" 
                />
                <label htmlFor="privacy" className="text-xs text-slate-600">
                  He leído y acepto el tratamiento de mis datos personales según el Aviso de Privacidad vigente.
                </label>
             </div>
             <p className="text-[10px] text-slate-500 ml-5">
               Tus datos se usan exclusivamente para la gestión operativa y de seguridad. Puedes ejercer tus derechos ARCO contactando a <b>privacidad@aerocrm.com</b>. 
               <button type="button" onClick={() => setShowPrivacyModal(true)} className="text-blue-600 font-medium ml-1 hover:underline">Leer Aviso Integral</button>.
             </p>
          </div>

          <button type="submit" className="w-full bg-slate-900 hover:bg-slate-800 text-white font-medium py-2.5 px-4 rounded-lg transition-colors shadow-md">
            Ingresar al Sistema
          </button>
        </form>
      </div>

      {showPrivacyModal && <PrivacyModal onClose={() => setShowPrivacyModal(false)} />}
    </div>
  );
}

                         <td className="p-3 text-slate-100">{log.accion}</td>
                       </tr>
                     ))}
                   </tbody>
                 </table>
               </div>
             </div>
          </div>
        )}

        {/* TAB: CONSOLA DB (JSON) */}
        {activeTab === 'db' && (
          <div className="flex-1 flex flex-col h-full overflow-hidden">
             <header className="bg-white shadow-sm border-b px-8 py-4 shrink-0">
               <h1 className="text-2xl font-bold text-slate-800">Consola de Base de Datos</h1>
               <p className="text-sm text-slate-500">Vista cruda del almacenamiento (Read-Only).</p>
             </header>
             <div className="flex-1 overflow-auto p-8 flex gap-4">
               <div className="flex-1 bg-slate-900 rounded-lg p-4 overflow-y-auto">
                 <h3 className="text-slate-400 font-bold mb-2">Tabla: CLIENTES</h3>
                 <pre className="text-emerald-400 font-mono text-xs">{JSON.stringify(clientes, null, 2)}</pre>
               </div>
               <div className="flex-1 bg-slate-900 rounded-lg p-4 overflow-y-auto">
                 <h3 className="text-slate-400 font-bold mb-2">Tabla: VENTAS</h3>
                 <pre className="text-emerald-400 font-mono text-xs">{JSON.stringify(ventas, null, 2)}</pre>
               </div>
             </div>
          </div>
        )}

        {/* TAB: PRIVACIDAD ARCO */}
        {activeTab === 'arco' && (
          <div className="flex-1 flex flex-col overflow-hidden">
             <header className="bg-white shadow-sm border-b px-8 py-4 shrink-0">
               <h1 className="text-2xl font-bold text-slate-800 flex items-center gap-2"><Shield className="text-blue-600"/> Centro de Cumplimiento ARCO</h1>
               <p className="text-sm text-slate-500">Gestión legal de datos personales a petición del titular.</p>
             </header>
             <div className="flex-1 overflow-auto p-8">
               <div className="bg-white rounded-xl shadow-sm border p-6 max-w-4xl">
                 <div className="bg-blue-50 border border-blue-100 p-4 rounded-lg mb-6 flex items-start gap-3">
                   <CheckCircle className="text-blue-600 mt-1 shrink-0"/>
                   <div className="text-sm text-blue-900">
                     <b>Protocolo Activo.</b> Como responsable del tratamiento, puedes atender aquí las solicitudes de Rectificación y Cancelación de los titulares. Todo movimiento quedará registrado en la bitácora de auditoría inmutable.
                   </div>
                 </div>

                 <h3 className="font-bold text-slate-700 mb-4">Búsqueda de Titular (Cliente)</h3>
                 <div className="flex gap-4 mb-6">
                   <input type="text" placeholder="Buscar por nombre o correo..." value={searchTerm} onChange={(e) => setSearchTerm(e.target.value)} className="flex-1 border p-2 rounded-lg" />
                 </div>

                 <div className="space-y-4">
                   {filteredClientes.map(c => (
                     <div key={c.id_pasajero} className="border p-4 rounded-lg flex justify-between items-center hover:shadow-md transition-shadow">
                        <div>
                          <p className="font-bold text-slate-800">{c.nombre} {c.apellido}</p>
                          <p className="text-sm text-slate-500">{c.correo_electronico} • ID: {c.id_pasajero}</p>
                        </div>
                        <div className="flex gap-2">
                           {canEdit && <button onClick={() => { setActiveTab('clientes'); setEditingClient({...c, documento_identidad: decryptData(c.documento_identidad_enc)}); setIsClientModalOpen(true); }} className="px-3 py-1.5 bg-slate-100 text-slate-700 rounded text-xs font-medium hover:bg-slate-200">Rectificar</button>}
                           {canDelete && <button onClick={() => { 
                             setClientes(clientes.filter(x => x.id_pasajero !== c.id_pasajero)); 
                             registrarLog(`EJECUTÓ DERECHO CANCELACIÓN ARCO: ID ${c.id_pasajero}`, 'PRIVACIDAD');
                           }} className="px-3 py-1.5 bg-red-100 text-red-700 rounded text-xs font-medium hover:bg-red-200 flex items-center gap-1"><Trash2 size={14}/> Purgar (Borrar)</button>}
                        </div>
                     </div>
                   ))}
                 </div>
               </div>
             </div>
          </div>
        )}

      </main>

      {/* Modales Compartidos */}
      {isClientModalOpen && (
        <ClientModal 
          client={editingClient} 
          onClose={() => setIsClientModalOpen(false)} 
          onSave={handleSaveClient} 
        />
      )}
      {isVentaModalOpen && (
        <VentaModal 
          clientes={clientes} 
          onClose={() => setIsVentaModalOpen(false)} 
          onSave={handleSaveVenta} 
        />
      )}
    </div>
  );
}

// --- MODAL DE CLIENTES (CRUD) ---
function ClientModal({ client, onClose, onSave }) {
  const [formData, setFormData] = useState({
    nombre: client?.nombre || '', apellido: client?.apellido || '', documento_identidad: client?.documento_identidad || '',
    correo_electronico: client?.correo_electronico || '', telefono: client?.telefono || ''
  });
  const [error, setError] = useState('');

  const handleSubmit = (e) => {
    e.preventDefault();
    if (Object.values(formData).some(val => val.trim() === '')) { setError('Todos los campos son obligatorios.'); return; }
    onSave(formData);
  };

  return (
    <div className="fixed inset-0 bg-slate-900/50 backdrop-blur-sm flex items-center justify-center z-50 p-4">
      <div className="bg-white rounded-xl shadow-2xl w-full max-w-lg overflow-hidden flex flex-col max-h-[95vh]">
        <div className="bg-slate-50 border-b px-6 py-4 flex justify-between shrink-0">
          <h3 className="font-bold text-slate-800 flex items-center gap-2"><Users size={20} className="text-blue-600"/> {client ? 'Editar Pasajero' : 'Registrar Pasajero'}</h3>
          <button onClick={onClose} className="text-slate-400 hover:text-slate-600">✕</button>
        </div>
        <form onSubmit={handleSubmit} className="p-6 overflow-y-auto">
          {error && <div className="bg-red-50 text-red-600 p-3 rounded mb-4 text-sm">{error}</div>}
          <div className="grid grid-cols-2 gap-4 mb-4">
            <div><label className="block text-sm font-medium text-slate-700 mb-1">Nombre</label><input type="text" value={formData.nombre} onChange={e => setFormData({...formData, nombre: e.target.value})} className="w-full px-3 py-2 border rounded outline-none focus:ring-2 focus:ring-blue-500" /></div>
            <div><label className="block text-sm font-medium text-slate-700 mb-1">Apellido</label><input type="text" value={formData.apellido} onChange={e => setFormData({...formData, apellido: e.target.value})} className="w-full px-3 py-2 border rounded outline-none focus:ring-2 focus:ring-blue-500" /></div>
          </div>
          <div className="mb-4">
            <label className="block text-sm font-medium text-slate-700 mb-1">Documento Identidad <Lock size={12} className="inline text-green-600"/></label>
            <input type="text" value={formData.documento_identidad} onChange={e => setFormData({...formData, documento_identidad: e.target.value})} className="w-full px-3 py-2 border rounded outline-none focus:ring-2 focus:ring-blue-500 uppercase" placeholder="Este dato será cifrado en BD" />
          </div>
          <div className="grid grid-cols-2 gap-4 mb-6">
            <div><label className="block text-sm font-medium text-slate-700 mb-1">Correo</label><input type="email" value={formData.correo_electronico} onChange={e => setFormData({...formData, correo_electronico: e.target.value})} className="w-full px-3 py-2 border rounded outline-none focus:ring-2 focus:ring-blue-500" /></div>
            <div><label className="block text-sm font-medium text-slate-700 mb-1">Teléfono</label><input type="tel" value={formData.telefono} onChange={e => setFormData({...formData, telefono: e.target.value})} className="w-full px-3 py-2 border rounded outline-none focus:ring-2 focus:ring-blue-500" /></div>
          </div>
          <div className="flex justify-end gap-3 pt-4 border-t">
            <button type="button" onClick={onClose} className="px-4 py-2 text-slate-600 hover:bg-slate-100 rounded">Cancelar</button>
            <button type="submit" className="px-4 py-2 bg-blue-600 text-white rounded shadow-sm hover:bg-blue-700">{client ? 'Actualizar' : 'Guardar'}</button>
          </div>
        </form>
      </div>
    </div>
  );
}

// --- MODAL DE VENTAS ---
function VentaModal({ clientes, onClose, onSave }) {
  const [formData, setFormData] = useState({ id_pasajero: '', vuelo: '', clase: 'Económica', monto: '' });
  const [error, setError] = useState('');
  
  const handleSubmit = (e) => {
    e.preventDefault();
    if (!formData.id_pasajero || !formData.vuelo || !formData.monto) {
      setError('Completa todos los campos obligatorios.');
      return;
    }
    onSave({ ...formData, id_pasajero: parseInt(formData.id_pasajero), monto: parseFloat(formData.monto) });
  };

  return (
    <div className="fixed inset-0 bg-slate-900/50 flex items-center justify-center z-50 p-4">
      <div className="bg-white rounded-lg w-full max-w-md overflow-hidden shadow-xl">
        <div className="px-5 py-3 border-b bg-green-50 flex justify-between items-center shrink-0">
          <h3 className="font-bold text-green-800">Nueva Venta de Boleto</h3>
          <button onClick={onClose} className="text-slate-500 hover:text-slate-800">✕</button>
        </div>
        <form onSubmit={handleSubmit} className="p-5 space-y-4">
          {error && <p className="text-red-500 text-xs bg-red-50 p-2 rounded border border-red-100">{error}</p>}
          <div>
             <label className="block text-sm font-medium text-slate-700 mb-1">Cliente Registrado</label>
             <select className="w-full border p-2 rounded text-sm focus:ring-2 focus:ring-green-500 outline-none" value={formData.id_pasajero} onChange={e => setFormData({...formData, id_pasajero: e.target.value})}>
               <option value="">-- Seleccionar --</option>
               {clientes.map(c => <option key={c.id_pasajero} value={c.id_pasajero}>{c.nombre} {c.apellido} (ID: {c.id_pasajero})</option>)}
             </select>
          </div>
          <div className="grid grid-cols-2 gap-3">
             <div>
               <label className="block text-sm font-medium text-slate-700 mb-1">Cód. Vuelo</label>
               <input type="text" className="w-full border p-2 rounded text-sm uppercase focus:ring-2 focus:ring-green-500 outline-none" placeholder="MX-123" value={formData.vuelo} onChange={e => setFormData({...formData, vuelo: e.target.value.toUpperCase()})} />
             </div>
             <div>
               <label className="block text-sm font-medium text-slate-700 mb-1">Monto ($)</label>
               <input type="number" className="w-full border p-2 rounded text-sm focus:ring-2 focus:ring-green-500 outline-none" placeholder="0.00" value={formData.monto} onChange={e => setFormData({...formData, monto: e.target.value})} />
             </div>
          </div>
          <div>
            <label className="block text-sm font-medium text-slate-700 mb-1">Clase</label>
            <select className="w-full border p-2 rounded text-sm focus:ring-2 focus:ring-green-500 outline-none" value={formData.clase} onChange={e => setFormData({...formData, clase: e.target.value})}>
              <option value="Económica">Económica</option>
              <option value="Ejecutiva">Ejecutiva</option>
              <option value="Primera Clase">Primera Clase</option>
            </select>
          </div>
          <div className="flex justify-end pt-2">
            <button type="submit" className="bg-green-600 text-white px-4 py-2 rounded shadow hover:bg-green-700 w-full font-medium">Procesar Venta</button>
          </div>
        </form>
      </div>
    </div>
  );
}
